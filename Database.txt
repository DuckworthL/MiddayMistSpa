-- ================================================================
-- SECTION 1: IDENTITY & SECURITY TABLES
-- ================================================================

-- Roles Table
CREATE TABLE Roles (
    RoleId INT IDENTITY(1,1) PRIMARY KEY,
    RoleCode VARCHAR(50) NOT NULL UNIQUE,
    RoleName NVARCHAR(100) NOT NULL,
    Description NVARCHAR(500),
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE()
);

-- Users Table (ASP.NET Identity compatible)
CREATE TABLE Users (
    UserId INT IDENTITY(1,1) PRIMARY KEY,
    Username NVARCHAR(256) NOT NULL UNIQUE,
    Email NVARCHAR(256) NOT NULL UNIQUE,
    EmailConfirmed BIT NOT NULL DEFAULT 0,
    PasswordHash NVARCHAR(MAX) NOT NULL,
    SecurityStamp NVARCHAR(MAX),
    PhoneNumber NVARCHAR(50),
    PhoneNumberConfirmed BIT NOT NULL DEFAULT 0,
    TwoFactorEnabled BIT NOT NULL DEFAULT 0,
    TotpSecretKey NVARCHAR(MAX),
    TwoFactorConfirmedAt DATETIME2 NULL,
    RecoveryCodes NVARCHAR(MAX),
    LockoutEnd DATETIME2 NULL,
    LockoutEnabled BIT NOT NULL DEFAULT 1,
    AccessFailedCount INT NOT NULL DEFAULT 0,
    RoleId INT NOT NULL,
    FirstName NVARCHAR(100) NOT NULL,
    LastName NVARCHAR(100) NOT NULL,
    IsActive BIT NOT NULL DEFAULT 1,
    LastLoginAt DATETIME2 NULL,
    PasswordExpiryDate DATETIME2 NOT NULL,
    MustChangePassword BIT NOT NULL DEFAULT 1,
    PasswordResetToken NVARCHAR(MAX) NULL,
    PasswordResetTokenExpiry DATETIME2 NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Users_Roles FOREIGN KEY (RoleId) REFERENCES Roles(RoleId)
);

-- Password History Table (prevents reuse of last 5 passwords)
CREATE TABLE PasswordHistory (
    PasswordHistoryId INT IDENTITY(1,1) PRIMARY KEY,
    UserId INT NOT NULL,
    PasswordHash NVARCHAR(MAX) NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_PasswordHistory_Users FOREIGN KEY (UserId) REFERENCES Users(UserId) ON DELETE CASCADE
);

-- User Sessions Table (track concurrent sessions, max 2 per user)
CREATE TABLE UserSessions (
    SessionId INT IDENTITY(1,1) PRIMARY KEY,
    UserId INT NOT NULL,
    Token NVARCHAR(MAX) NOT NULL,
    IpAddress NVARCHAR(50),
    UserAgent NVARCHAR(500),
    IsActive BIT NOT NULL DEFAULT 1,
    ExpiresAt DATETIME2 NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_UserSessions_Users FOREIGN KEY (UserId) REFERENCES Users(UserId) ON DELETE CASCADE
);

-- Audit Logs Table
CREATE TABLE AuditLogs (
    AuditLogId BIGINT IDENTITY(1,1) PRIMARY KEY,
    UserId INT NULL,
    Action NVARCHAR(100) NOT NULL, -- Login, Logout, Create, Update, Delete, etc.
    TableName NVARCHAR(100),
    RecordId NVARCHAR(50),
    OldValues NVARCHAR(MAX), -- JSON format
    NewValues NVARCHAR(MAX), -- JSON format
    IpAddress NVARCHAR(50),
    UserAgent NVARCHAR(500),
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_AuditLogs_Users FOREIGN KEY (UserId) REFERENCES Users(UserId) ON DELETE SET NULL
);

-- Employees Table
CREATE TABLE Employees (
    EmployeeId INT IDENTITY(1,1) PRIMARY KEY,
    UserId INT NULL, -- Links to Users table if employee has login access
    EmployeeCode NVARCHAR(50) NOT NULL UNIQUE,
    FirstName NVARCHAR(100) NOT NULL,
    LastName NVARCHAR(100) NOT NULL,
    MiddleName NVARCHAR(100),
    DateOfBirth DATE NOT NULL,
    Gender NVARCHAR(10) NOT NULL, -- Male, Female, Other
    CivilStatus NVARCHAR(20), -- Single, Married, Widowed, Divorced
    Address NVARCHAR(500),
    City NVARCHAR(100),
    Province NVARCHAR(100),
    PostalCode NVARCHAR(20),
    PhoneNumber NVARCHAR(50) NOT NULL,
    Email NVARCHAR(256),
    EmergencyContactName NVARCHAR(200),
    EmergencyContactPhone NVARCHAR(50),
    
    -- Philippine Government IDs
    SSSNumber NVARCHAR(20),
    PhilHealthNumber NVARCHAR(20),
    PagIBIGNumber NVARCHAR(20),
    TINNumber NVARCHAR(20),
    
    -- Employment Details
    Position NVARCHAR(100) NOT NULL,
    Department NVARCHAR(100),
    HireDate DATE NOT NULL,
    EmploymentType NVARCHAR(50) NOT NULL, -- Regular, Contractual, Part-Time, Probationary
    EmploymentStatus NVARCHAR(50) NOT NULL DEFAULT 'Active', -- Active, Resigned, Terminated, On Leave
    
    -- Salary Information
    DailyRate DECIMAL(18,2) NOT NULL,
    MonthlyBasicSalary DECIMAL(18,2) NOT NULL,
    PayrollType NVARCHAR(50) NOT NULL DEFAULT 'Semi-Monthly', -- Semi-Monthly, Monthly, Daily
    
    -- Therapist Specific
    IsTherapist BIT NOT NULL DEFAULT 0,
    Specialization NVARCHAR(200), -- Swedish Massage, Deep Tissue, Reflexology, etc.
    LicenseNumber NVARCHAR(100),
    LicenseExpiryDate DATE,
    
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Employees_Users FOREIGN KEY (UserId) REFERENCES Users(UserId) ON DELETE SET NULL
);

-- Employee Schedules Table (weekly schedule template)
CREATE TABLE EmployeeSchedules (
    ScheduleId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    DayOfWeek INT NOT NULL, -- 0=Sunday, 1=Monday, ..., 6=Saturday
    ShiftStartTime TIME NOT NULL,
    ShiftEndTime TIME NOT NULL,
    BreakStartTime TIME,
    BreakEndTime TIME,
    IsRestDay BIT NOT NULL DEFAULT 0,
    EffectiveDate DATE NOT NULL,
    EndDate DATE NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_EmployeeSchedules_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE
);

-- Employee Shifts Table (recurring weekly shift definitions)
CREATE TABLE EmployeeShifts (
    ShiftId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    DayOfWeek INT NOT NULL, -- 0=Sunday, 1=Monday, ..., 6=Saturday
    StartTime TIME NOT NULL,
    EndTime TIME NOT NULL,
    IsRecurring BIT NOT NULL DEFAULT 1,
    EffectiveFrom DATETIME2 NOT NULL,
    EffectiveTo DATETIME2 NULL,
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_EmployeeShifts_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE
);

-- Shift Exceptions Table (overrides to recurring shifts: time off, sick leave, custom hours)
CREATE TABLE ShiftExceptions (
    ExceptionId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    ExceptionDate DATETIME2 NOT NULL,
    ExceptionType NVARCHAR(50) NOT NULL, -- TimeOff, SickLeave, Emergency, CustomHours
    StartTime TIME NULL, -- NULL if full day off
    EndTime TIME NULL, -- NULL if full day off
    Reason NVARCHAR(500),
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_ShiftExceptions_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE
);

-- Attendance Records Table (daily clock-in/out tracking)
CREATE TABLE AttendanceRecords (
    AttendanceId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    Date DATETIME2 NOT NULL,
    ClockIn DATETIME2 NULL,
    ClockOut DATETIME2 NULL,
    BreakStart DATETIME2 NULL,
    BreakEnd DATETIME2 NULL,
    TotalHours DECIMAL(18,2) NOT NULL DEFAULT 0,
    BreakMinutes DECIMAL(18,2) NOT NULL DEFAULT 0,
    Status NVARCHAR(50) NOT NULL DEFAULT 'ClockedIn', -- ClockedIn, OnBreak, ClockedOut, Absent
    IsApproved BIT NOT NULL DEFAULT 0,
    ClockedInByUserId INT NULL, -- Null = self, otherwise admin/HR who clocked them in
    Notes NVARCHAR(1000),
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_AttendanceRecords_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE
);

-- Employee Time Off Requests
CREATE TABLE TimeOffRequests (
    TimeOffRequestId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    LeaveType NVARCHAR(50) NOT NULL, -- SIL, Sick Leave, Maternity, Paternity, Solo Parent, Emergency
    StartDate DATE NOT NULL,
    EndDate DATE NOT NULL,
    TotalDays DECIMAL(5,2) NOT NULL,
    Reason NVARCHAR(500),
    Status NVARCHAR(50) NOT NULL DEFAULT 'Pending', -- Pending, Approved, Rejected
    ApprovedBy INT NULL,
    ApprovedAt DATETIME2 NULL,
    RejectionReason NVARCHAR(500),
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_TimeOffRequests_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE,
    CONSTRAINT FK_TimeOffRequests_ApprovedBy FOREIGN KEY (ApprovedBy) REFERENCES Users(UserId) ON DELETE SET NULL
);

-- Employee Leave Balances
CREATE TABLE EmployeeLeaveBalances (
    LeaveBalanceId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    Year INT NOT NULL,
    SILDays DECIMAL(5,2) NOT NULL DEFAULT 5.0, -- Service Incentive Leave (5 days/year in PH)
    SILUsed DECIMAL(5,2) NOT NULL DEFAULT 0,
    SickLeaveDays DECIMAL(5,2) NOT NULL DEFAULT 0,
    SickLeaveUsed DECIMAL(5,2) NOT NULL DEFAULT 0,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_EmployeeLeaveBalances_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE,
    CONSTRAINT UQ_EmployeeLeaveBalances UNIQUE (EmployeeId, Year)
);

-- Employee Advances/Loans
CREATE TABLE EmployeeAdvances (
    AdvanceId INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeId INT NOT NULL,
    AdvanceType NVARCHAR(50) NOT NULL, -- Cash Advance, SSS Loan, Pag-IBIG Loan
    Amount DECIMAL(18,2) NOT NULL,
    Balance DECIMAL(18,2) NOT NULL,
    MonthlyDeduction DECIMAL(18,2) NOT NULL,
    StartDate DATE NOT NULL,
    NumberOfInstallments INT NOT NULL,
    InstallmentsPaid INT NOT NULL DEFAULT 0,
    Status NVARCHAR(50) NOT NULL DEFAULT 'Active', -- Active, Fully Paid
    ApprovedBy INT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_EmployeeAdvances_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE,
    CONSTRAINT FK_EmployeeAdvances_ApprovedBy FOREIGN KEY (ApprovedBy) REFERENCES Users(UserId) ON DELETE SET NULL
);

-- Payroll Periods
CREATE TABLE PayrollPeriods (

    PeriodName NVARCHAR(100) NOT NULL, -- e.g., "January 1-15, 2026"

    EndDate DATE NOT NULL,
    PayrollType NVARCHAR(50) NOT NULL, -- Semi-Monthly, Monthly
    CutoffDate DATE NOT NULL,
    PaymentDate DATE NOT NULL,

    HoursWorked DECIMAL(8,2) NOT NULL,
    OvertimeHours DECIMAL(8,2) NOT NULL DEFAULT 0,
    NightDifferentialHours DECIMAL(8,2) NOT NULL DEFAULT 0,

    BasicSalary DECIMAL(18,2) NOT NULL,

    HolidayPay DECIMAL(18,2) NOT NULL DEFAULT 0,
    RestDayPay DECIMAL(18,2) NOT NULL DEFAULT 0,
    Commissions DECIMAL(18,2) NOT NULL DEFAULT 0,
    Tips DECIMAL(18,2) NOT NULL DEFAULT 0,
    RiceAllowance DECIMAL(18,2) NOT NULL DEFAULT 0,
    LaundryAllowance DECIMAL(18,2) NOT NULL DEFAULT 0,

    -- Mandatory Deductions - Employee Share (Philippine)
    SSSContribution DECIMAL(18,2) NOT NULL DEFAULT 0,
    PhilHealthContribution DECIMAL(18,2) NOT NULL DEFAULT 0,
    PagIBIGContribution DECIMAL(18,2) NOT NULL DEFAULT 0,
    WithholdingTax DECIMAL(18,2) NOT NULL DEFAULT 0,
    
    -- Mandatory Deductions - Employer Share (for remittance reports)

    Tardiness DECIMAL(18,2) NOT NULL DEFAULT 0,
    Absences DECIMAL(18,2) NOT NULL DEFAULT 0,
    CashAdvances DECIMAL(18,2) NOT NULL DEFAULT 0,
    LoanDeductions DECIMAL(18,2) NOT NULL DEFAULT 0,
    OtherDeductions DECIMAL(18,2) NOT NULL DEFAULT 0,
    TotalDeductions DECIMAL(18,2) NOT NULL,
    
    -- Net Pay
    NetPay DECIMAL(18,2) NOT NULL,
    
    -- Payment Details
    PaymentMethod NVARCHAR(50), -- Cash, Bank Transfer
    PaymentDate DATE,
    PaymentStatus NVARCHAR(50) NOT NULL DEFAULT 'Pending', -- Pending, Paid
    
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_PayrollRecords_PayrollPeriods FOREIGN KEY (PayrollPeriodId) REFERENCES PayrollPeriods(PayrollPeriodId) ON DELETE CASCADE,
    CONSTRAINT FK_PayrollRecords_Employees FOREIGN KEY (EmployeeId) REFERENCES Employees(EmployeeId) ON DELETE CASCADE
);

-- Customers Table
CREATE TABLE Customers (
    CustomerId INT IDENTITY(1,1) PRIMARY KEY,
    CustomerCode NVARCHAR(50) NOT NULL UNIQUE,
    FirstName NVARCHAR(100) NOT NULL,
    LastName NVARCHAR(100) NOT NULL,
    Email NVARCHAR(256),
    PhoneNumber NVARCHAR(50) NOT NULL,
    DateOfBirth DATE,
    Gender NVARCHAR(10),
    Address NVARCHAR(500),
    City NVARCHAR(100),
    Province NVARCHAR(100),
    PostalCode NVARCHAR(20),
    
    -- Membership

    MembershipStartDate DATE,
    MembershipExpiryDate DATE,
    LoyaltyPoints INT NOT NULL DEFAULT 0,
    
    -- Preferences
    PreferredTherapistId INT NULL,
    PressurePreference NVARCHAR(50), -- Light, Medium, Firm
    TemperaturePreference NVARCHAR(50), -- Cool, Warm, Hot
    MusicPreference NVARCHAR(50), -- Classical, Spa Music, Nature Sounds, Silence
    Allergies NVARCHAR(500), -- IMPORTANT: Red alert in UI
    MedicalNotes NVARCHAR(1000),
    SpecialRequests NVARCHAR(500),
    
    -- Emergency Contact
    EmergencyContactName NVARCHAR(200),
    EmergencyContactPhone NVARCHAR(50),
    EmergencyContactRelationship NVARCHAR(100),
    
    -- Communication Preferences

    LastVisitDate DATE,
    TotalVisits INT NOT NULL DEFAULT 0,
    TotalSpent DECIMAL(18,2) NOT NULL DEFAULT 0,

    CustomerSegment NVARCHAR(50), -- VIP Platinum, Loyal Regulars, At-Risk, etc.

    IsActive BIT NOT NULL DEFAULT 1,

);

-- Customer Segments Table (DBSCAN results)
CREATE TABLE CustomerSegments (
    SegmentId INT IDENTITY(1,1) PRIMARY KEY,
    SegmentName NVARCHAR(100) NOT NULL,
    SegmentCode NVARCHAR(50) NOT NULL UNIQUE,
    Description NVARCHAR(500),
    ClusterId INT NOT NULL, -- DBSCAN cluster ID (-1 for noise/outliers)
    AverageRecency DECIMAL(10,2), -- Days since last visit
    AverageFrequency DECIMAL(10,2), -- Visits per month
    AverageMonetaryValue DECIMAL(18,2), -- Lifetime value
    CustomerCount INT NOT NULL DEFAULT 0,
    RecommendedAction NVARCHAR(500),
    LastAnalysisDate DATETIME2,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE()
);

-- Loyalty Point Transactions Table (batch-level tracking with expiry)
CREATE TABLE LoyaltyPointTransactions (
    LoyaltyPointTransactionId INT IDENTITY(1,1) PRIMARY KEY,
    CustomerId INT NOT NULL,
    TransactionType NVARCHAR(50) NOT NULL, -- Earn, Redeem, Expire, Adjust
    Points INT NOT NULL, -- Positive for earn/adjust, negative for redeem/expire
    BalanceRemaining INT NOT NULL, -- Remaining points from this batch (for expiry tracking)
    EarnedDate DATETIME2 NOT NULL DEFAULT GETDATE(),
    ExpiryDate DATETIME2 NULL, -- 12 months from earned date; null for non-earn types
    TransactionId INT NULL, -- Links to POS transaction (null for manual adjustments/expiry)
    Description NVARCHAR(500),
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_LoyaltyPointTransactions_Customers FOREIGN KEY (CustomerId) REFERENCES Customers(CustomerId) ON DELETE CASCADE,
    CONSTRAINT FK_LoyaltyPointTransactions_Transactions FOREIGN KEY (TransactionId) REFERENCES Transactions(TransactionId)
);

    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE()

-- Products/Inventory Table
CREATE TABLE Products (
    ProductId INT IDENTITY(1,1) PRIMARY KEY,
    ProductCategoryId INT NOT NULL,
    ProductCode NVARCHAR(50) NOT NULL UNIQUE,
    ProductName NVARCHAR(200) NOT NULL,
    Description NVARCHAR(1000),
    ProductType NVARCHAR(50) NOT NULL, -- Retail, Supply, Consumable
    
    -- Inventory
    CurrentStock DECIMAL(18,3) NOT NULL DEFAULT 0,
    ReorderLevel DECIMAL(18,3) NOT NULL,
    UnitOfMeasure NVARCHAR(20) NOT NULL, -- pcs, ml, g, bottles, etc.
    
    -- Pricing
    CostPrice DECIMAL(18,2) NOT NULL,
    SellingPrice DECIMAL(18,2),
    RetailCommissionRate DECIMAL(5,4) DEFAULT 0.10, -- 10% for retail products
    
    -- Tracking
    ExpiryDate DATE,
    Supplier NVARCHAR(200),
    
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Products_ProductCategories FOREIGN KEY (ProductCategoryId) REFERENCES ProductCategories(ProductCategoryId)

-- Product Batches Table (batch/lot tracking for FIFO costing)
CREATE TABLE ProductBatches (
    ProductBatchId INT IDENTITY(1,1) PRIMARY KEY,
    ProductId INT NOT NULL,
    BatchNumber NVARCHAR(100) NOT NULL,
    CostPrice DECIMAL(18,2) NOT NULL,
    QuantityReceived DECIMAL(18,3) NOT NULL,
    QuantityRemaining DECIMAL(18,3) NOT NULL,
    ExpiryDate DATETIME2 NULL,
    ReceivedDate DATETIME2 NOT NULL DEFAULT GETDATE(),
    PurchaseOrderItemId INT NULL,
    SupplierId INT NULL,
    Notes NVARCHAR(500),
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_ProductBatches_Products FOREIGN KEY (ProductId) REFERENCES Products(ProductId) ON DELETE CASCADE,
    CONSTRAINT FK_ProductBatches_POItems FOREIGN KEY (PurchaseOrderItemId) REFERENCES PurchaseOrderItems(POItemId),
    CONSTRAINT FK_ProductBatches_Suppliers FOREIGN KEY (SupplierId) REFERENCES Suppliers(SupplierId)
);

-- Service-Product Requirements (Many-to-Many)
CREATE TABLE ServiceProductRequirements (
    RequirementId INT IDENTITY(1,1) PRIMARY KEY,
    ServiceId INT NOT NULL,
    ProductId INT NOT NULL,
    QuantityRequired DECIMAL(18,3) NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_ServiceProductReq_Services FOREIGN KEY (ServiceId) REFERENCES Services(ServiceId) ON DELETE CASCADE,
    CONSTRAINT FK_ServiceProductReq_Products FOREIGN KEY (ProductId) REFERENCES Products(ProductId) ON DELETE CASCADE,
    CONSTRAINT UQ_ServiceProduct UNIQUE (ServiceId, ProductId)

-- Suppliers Table
CREATE TABLE Suppliers (
    SupplierId INT IDENTITY(1,1) PRIMARY KEY,
    SupplierCode NVARCHAR(50) NOT NULL UNIQUE,
    SupplierName NVARCHAR(200) NOT NULL,
    ContactPerson NVARCHAR(100),
    Email NVARCHAR(256),
    PhoneNumber NVARCHAR(50),
    Address NVARCHAR(500),
    City NVARCHAR(100),
    Province NVARCHAR(100),
    TINNumber NVARCHAR(20),
    PaymentTerms NVARCHAR(100), -- Net 30, Net 60, COD, etc.
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE()
);

-- Purchase Orders
CREATE TABLE PurchaseOrders (
    PurchaseOrderId INT IDENTITY(1,1) PRIMARY KEY,
    PONumber NVARCHAR(50) NOT NULL UNIQUE,
    SupplierId INT NOT NULL,
    OrderDate DATE NOT NULL,
    ExpectedDeliveryDate DATE,
    TotalAmount DECIMAL(18,2) NOT NULL,
    Status NVARCHAR(50) NOT NULL DEFAULT 'Pending', -- Pending, Approved, Delivered, Cancelled
    ApprovedBy INT NULL,
    ApprovedAt DATETIME2 NULL,
    ReceivedDate DATETIME2 NULL,
    Notes NVARCHAR(1000),
    CreatedBy INT NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_PurchaseOrders_Suppliers FOREIGN KEY (SupplierId) REFERENCES Suppliers(SupplierId),
    CONSTRAINT FK_PurchaseOrders_ApprovedBy FOREIGN KEY (ApprovedBy) REFERENCES Users(UserId) ON DELETE SET NULL,
    CONSTRAINT FK_PurchaseOrders_CreatedBy FOREIGN KEY (CreatedBy) REFERENCES Users(UserId)
);

-- Purchase Order Items
CREATE TABLE PurchaseOrderItems (
    POItemId INT IDENTITY(1,1) PRIMARY KEY,
    PurchaseOrderId INT NOT NULL,
    ProductId INT NOT NULL,
    Quantity DECIMAL(18,3) NOT NULL,
    UnitPrice DECIMAL(18,2) NOT NULL,
    TotalPrice DECIMAL(18,2) NOT NULL,
    QuantityReceived DECIMAL(18,3) NOT NULL DEFAULT 0,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_POItems_PurchaseOrders FOREIGN KEY (PurchaseOrderId) REFERENCES PurchaseOrders(PurchaseOrderId) ON DELETE CASCADE,
    CONSTRAINT FK_POItems_Products FOREIGN KEY (ProductId) REFERENCES Products(ProductId)
);

-- Stock Adjustments
CREATE TABLE StockAdjustments (
    AdjustmentId INT IDENTITY(1,1) PRIMARY KEY,
    ProductId INT NOT NULL,
    AdjustmentType NVARCHAR(50) NOT NULL, -- Received, Sold, Damaged, Expired, Service Usage, Audit
    QuantityBefore DECIMAL(18,3) NOT NULL,
    QuantityChange DECIMAL(18,3) NOT NULL, -- Positive or negative
    QuantityAfter DECIMAL(18,3) NOT NULL,
    Reason NVARCHAR(500),
    ReferenceNumber NVARCHAR(100), -- Links to PO, Invoice, Appointment, etc.
    AdjustedBy INT NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_StockAdjustments_Products FOREIGN KEY (ProductId) REFERENCES Products(ProductId),
    CONSTRAINT FK_StockAdjustments_AdjustedBy FOREIGN KEY (AdjustedBy) REFERENCES Users(UserId)
);

-- ================================================================
-- SECTION 6: APPOINTMENTS & BOOKINGS TABLES
-- ================================================================

-- Rooms Table (treatment rooms/service areas)
CREATE TABLE Rooms (
    RoomId INT IDENTITY(1,1) PRIMARY KEY,
    RoomName NVARCHAR(100) NOT NULL,
    RoomCode NVARCHAR(50) NOT NULL,
    Description NVARCHAR(500),
    RoomType NVARCHAR(50) NOT NULL DEFAULT 'Treatment', -- Treatment, Massage, Facial, Nail, Multi-Purpose
    Capacity INT NOT NULL DEFAULT 1,
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE()
);

-- Appointments Table
CREATE TABLE Appointments (
    AppointmentId INT IDENTITY(1,1) PRIMARY KEY,
    AppointmentNumber NVARCHAR(50) NOT NULL UNIQUE,
    CustomerId INT NOT NULL,
    ServiceId INT NOT NULL,
    TherapistId INT NULL,
    RoomId INT NULL,
    
    -- Scheduling
    AppointmentDate DATE NOT NULL,
    StartTime TIME NOT NULL,
    EndTime TIME NOT NULL,
    
    -- Status Tracking
    -- Scheduled -> Confirmed -> Checked-In -> In Progress -> Completed
    --                        -> Cancelled
    --                        -> No-Show
    Status NVARCHAR(50) NOT NULL DEFAULT 'Scheduled',
    
    -- Booking Source
    BookingSource NVARCHAR(50) DEFAULT 'Direct', -- Direct, Walk-In, Phone, Email
    
    -- Service Notes
    CustomerNotes NVARCHAR(1000),
    TherapistNotes NVARCHAR(1000),
    
    -- Timestamps
    CheckedInAt DATETIME2,
    ServiceStartedAt DATETIME2,
    ServiceCompletedAt DATETIME2,
    CancelledAt DATETIME2,
    CancellationReason NVARCHAR(500),
    
    -- Archive flag
    IsArchived BIT NOT NULL DEFAULT 0,
    ArchivedAt DATETIME2 NULL,
    
    CreatedBy INT,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Appointments_Customers FOREIGN KEY (CustomerId) REFERENCES Customers(CustomerId),
    CONSTRAINT FK_Appointments_Services FOREIGN KEY (ServiceId) REFERENCES Services(ServiceId),
    CONSTRAINT FK_Appointments_Therapists FOREIGN KEY (TherapistId) REFERENCES Employees(EmployeeId) ON DELETE SET NULL,
    CONSTRAINT FK_Appointments_Rooms FOREIGN KEY (RoomId) REFERENCES Rooms(RoomId) ON DELETE SET NULL,
    CONSTRAINT FK_Appointments_CreatedBy FOREIGN KEY (CreatedBy) REFERENCES Users(UserId) ON DELETE SET NULL
);

-- Appointment Service Items (multi-service support per appointment)
CREATE TABLE AppointmentServiceItems (
    AppointmentServiceItemId INT IDENTITY(1,1) PRIMARY KEY,
    AppointmentId INT NOT NULL,
    ServiceId INT NOT NULL,
    UnitPrice DECIMAL(18,2) NOT NULL, -- Price snapshot at booking time
    DurationMinutes INT NOT NULL, -- Duration snapshot from service
    Quantity INT NOT NULL DEFAULT 1,
    AddedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_AppointmentServiceItems_Appointments FOREIGN KEY (AppointmentId) REFERENCES Appointments(AppointmentId) ON DELETE CASCADE,
    CONSTRAINT FK_AppointmentServiceItems_Services FOREIGN KEY (ServiceId) REFERENCES Services(ServiceId)
);

-- Waitlist Table (customers waiting for appointment slots)
CREATE TABLE Waitlists (
    WaitlistId INT IDENTITY(1,1) PRIMARY KEY,
    CustomerId INT NOT NULL,
    ServiceId INT NOT NULL,
    PreferredTherapistId INT NULL,
    PreferredDate DATETIME2 NULL,
    PreferredTimeFrom TIME NULL,
    PreferredTimeTo TIME NULL,
    Status NVARCHAR(50) NOT NULL DEFAULT 'Waiting', -- Waiting, Notified, Booked, Expired, Cancelled
    Notes NVARCHAR(1000),
    NotifiedAt DATETIME2 NULL,
    ConvertedAppointmentId INT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Waitlists_Customers FOREIGN KEY (CustomerId) REFERENCES Customers(CustomerId),
    CONSTRAINT FK_Waitlists_Services FOREIGN KEY (ServiceId) REFERENCES Services(ServiceId),
    CONSTRAINT FK_Waitlists_PreferredTherapist FOREIGN KEY (PreferredTherapistId) REFERENCES Employees(EmployeeId) ON DELETE SET NULL,
    CONSTRAINT FK_Waitlists_ConvertedAppointment FOREIGN KEY (ConvertedAppointmentId) REFERENCES Appointments(AppointmentId)
);

-- ================================================================
-- SECTION 7: POINT OF SALE & TRANSACTIONS TABLES
-- ================================================================

-- Transactions Table (POS)
CREATE TABLE Transactions (
    TransactionId INT IDENTITY(1,1) PRIMARY KEY,
    TransactionNumber NVARCHAR(50) NOT NULL UNIQUE,
    CustomerId INT NULL, -- Nullable for walk-in customers
    AppointmentId INT NULL, -- Links to appointment if checkout after service
    
    -- Amounts (in PHP)
    Subtotal DECIMAL(18,2) NOT NULL,
    DiscountAmount DECIMAL(18,2) NOT NULL DEFAULT 0,
    DiscountPercentage DECIMAL(5,2) NOT NULL DEFAULT 0,
    TaxAmount DECIMAL(18,2) NOT NULL DEFAULT 0, -- 12% VAT
    TipAmount DECIMAL(18,2) NOT NULL DEFAULT 0,
    TotalAmount DECIMAL(18,2) NOT NULL,
    
    -- Payment
    PaymentMethod NVARCHAR(50) NOT NULL, -- Cash, Card, Split
    PaymentStatus NVARCHAR(50) NOT NULL DEFAULT 'Pending', -- Pending, Paid, Refunded, Voided
    AmountTendered DECIMAL(18,2), -- Cash amount given by customer
    ChangeAmount DECIMAL(18,2), -- Change returned to customer
    
    -- Loyalty
    LoyaltyPointsEarned INT NOT NULL DEFAULT 0,
    
    -- Multi-Currency Support
    ClientCurrency NVARCHAR(10) NOT NULL DEFAULT 'PHP', -- Client's preferred currency
    ClientCountryCode NVARCHAR(10), -- From IpWhoIs (PH, US, JP, etc.)
    ClientIPAddress NVARCHAR(50), -- Client IP for geolocation
    ExchangeRate DECIMAL(18,6) NOT NULL DEFAULT 1.0, -- 1 PHP = X target currency
    TotalInClientCurrency DECIMAL(18,2), -- Total converted to client currency
    
    -- Tracking
    TransactionDate DATETIME2 NOT NULL DEFAULT GETDATE(),
    VoidedAt DATETIME2 NULL,
    VoidedBy INT NULL,
    VoidReason NVARCHAR(500),
    
    CashierId INT NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    
    -- Concurrency control
    RowVersion ROWVERSION NOT NULL,

    CONSTRAINT FK_Transactions_Customers FOREIGN KEY (CustomerId) REFERENCES Customers(CustomerId),
    CONSTRAINT FK_Transactions_Appointments FOREIGN KEY (AppointmentId) REFERENCES Appointments(AppointmentId),
    CONSTRAINT FK_Transactions_Cashier FOREIGN KEY (CashierId) REFERENCES Users(UserId),
    CONSTRAINT FK_Transactions_VoidedBy FOREIGN KEY (VoidedBy) REFERENCES Users(UserId)
);

-- Transaction Items (Services)
CREATE TABLE TransactionServiceItems (
    TransactionServiceItemId INT IDENTITY(1,1) PRIMARY KEY,
    TransactionId INT NOT NULL,
    ServiceId INT NOT NULL,
    TherapistId INT NULL,
    Quantity INT NOT NULL DEFAULT 1,
    UnitPrice DECIMAL(18,2) NOT NULL,
    TotalPrice DECIMAL(18,2) NOT NULL,
    CommissionRate DECIMAL(5,4) NOT NULL,
    CommissionAmount DECIMAL(18,2) NOT NULL,
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_TransactionServiceItems_Transactions FOREIGN KEY (TransactionId) REFERENCES Transactions(TransactionId) ON DELETE CASCADE,
    CONSTRAINT FK_TransactionServiceItems_Services FOREIGN KEY (ServiceId) REFERENCES Services(ServiceId),
    CONSTRAINT FK_TransactionServiceItems_Therapists FOREIGN KEY (TherapistId) REFERENCES Employees(EmployeeId) ON DELETE SET NULL
);

-- Transaction Items (Products/Retail)
CREATE TABLE TransactionProductItems (
    TransactionProductItemId INT IDENTITY(1,1) PRIMARY KEY,
    TransactionId INT NOT NULL,
    ProductId INT NOT NULL,
    Quantity DECIMAL(18,3) NOT NULL,
    UnitPrice DECIMAL(18,2) NOT NULL,
    TotalPrice DECIMAL(18,2) NOT NULL,
    CommissionRate DECIMAL(5,4) NOT NULL DEFAULT 0.10, -- 10% default for retail
    CommissionAmount DECIMAL(18,2) NOT NULL,
    SoldBy INT NULL, -- Employee who sold the product
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_TransactionProductItems_Transactions FOREIGN KEY (TransactionId) REFERENCES Transactions(TransactionId) ON DELETE CASCADE,
    CONSTRAINT FK_TransactionProductItems_Products FOREIGN KEY (ProductId) REFERENCES Products(ProductId),
    CONSTRAINT FK_TransactionProductItems_SoldBy FOREIGN KEY (SoldBy) REFERENCES Users(UserId)

-- Refunds Table
CREATE TABLE Refunds (
    RefundId INT IDENTITY(1,1) PRIMARY KEY,
    TransactionId INT NOT NULL,
    RefundAmount DECIMAL(18,2) NOT NULL,
    RefundMethod NVARCHAR(50) NOT NULL, -- Cash, Card Reversal
    Reason NVARCHAR(500),
    RefundType NVARCHAR(50) NOT NULL, -- Full, Partial
    ApprovedBy INT NOT NULL,
    ProcessedBy INT NOT NULL,
    RefundDate DATETIME2 NOT NULL DEFAULT GETDATE(),
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Refunds_Transactions FOREIGN KEY (TransactionId) REFERENCES Transactions(TransactionId) ON DELETE CASCADE,
    CONSTRAINT FK_Refunds_ApprovedBy FOREIGN KEY (ApprovedBy) REFERENCES Users(UserId),
    CONSTRAINT FK_Refunds_ProcessedBy FOREIGN KEY (ProcessedBy) REFERENCES Users(UserId)
);

-- Journal Entries
CREATE TABLE JournalEntries (
    JournalEntryId INT IDENTITY(1,1) PRIMARY KEY,
    EntryNumber NVARCHAR(50) NOT NULL,
    EntryDate DATETIME2 NOT NULL,
    Description NVARCHAR(1000),
    ReferenceType NVARCHAR(50), -- Transaction, Payroll, PO, Adjustment
    ReferenceId NVARCHAR(50),
    TotalDebit DECIMAL(18,2) NOT NULL,
    TotalCredit DECIMAL(18,2) NOT NULL,
    Status NVARCHAR(50) NOT NULL DEFAULT 'Draft', -- Draft, Posted, Voided
